function getDir(path) {
  let node = vfs;
  for (let i = 1; i < path.length; i++) {
    const c = node.children && node.children.find(x => x.name === path[i]);
    if (!c || c.type !== 'directory') return null;
    node = c;
  }
  return node;
}

function handleCommand(input) {
  const trimmed = input.trim();
  if (!trimmed) return;
  termHistory.unshift(trimmed);
  historyIndex = -1;

  termPrint(`cyperuser@cyperweb-os:${currentPath.join('/')}$ ${trimmed}`, 'term-line-user');

  const parts = trimmed.split(' ');
  const cmd = parts[0].toLowerCase();
  const args = parts.slice(1);

  switch (cmd) {
    case 'help':
      termPrint(`Available commands:

  ── Navigation ──────────────────────────────
  pwd                   - Print working directory
  ls                    - List files & folders
  ls -a                 - Show hidden files too
  ls -l                 - Long format (perms + size)
  cd <dir>              - Enter folder
  cd ..                 - Go back one level
  cd /                  - Go to root
  cd ~                  - Go to home

  ── Directories ─────────────────────────────
  mkdir <name>          - Create folder
  mkdir -p <a/b/c>      - Create nested folders
  rmdir <name>          - Remove empty folder
  rm -d <name>          - Remove empty folder

  ── Files ───────────────────────────────────
  touch <file>          - Create file
  touch -a <file>       - Update last access time
  touch -m <file>       - Update last modified time
  touch -c <file>       - Update time (no create)
  touch -d "date" <f>   - Set specific date on file
  touch -r <f1> <f2>    - Copy timestamp f1 → f2
  stat <file>           - Show full file details
  cat <file>            - Display file contents
  echo "text"           - Print text
  echo "text" > <file>  - Write text into file

  ── Remove ──────────────────────────────────
  rm <file>             - Remove file
  rm -i <file>          - Remove with confirmation
  rm -v <file>          - Remove with message
  rm -iv <file>         - Remove with confirm + msg
  rm -r <folder>        - Remove folder recursively
  rm -ir <folder>       - Remove folder with confirm

  ── System ──────────────────────────────────
  whoami                - Show current user
  ps                    - List processes
  free                  - Memory info
  uname                 - System info
  neofetch              - System info with logo
  history               - Show command history
  clear                 - Clear terminal
  cyperWeb              - CYPER WEB greeting
  help                  - Show this help`);
      break;

    case 'ls': {
      const dir = getCurrentDir();
      if (!dir) { termPrint('ls: no such directory', 'term-line-error'); break; }
      const showHidden = args.includes('-a');
      const longFormat = args.includes('-l');
      let items = dir.children || [];
      // hidden files start with '.' — add virtual . and .. when -a
      let allItems = showHidden
        ? [{ name: '.', type: 'directory' }, { name: '..', type: 'directory' }, ...items]
        : items;
      if (allItems.length === 0) { termPrint('(empty directory)'); break; }
      if (longFormat) {
        const now = new Date();
        const dateStr = now.toLocaleDateString('en-GB', { day:'2-digit', month:'short' });
        const lines = allItems.map(c => {
          const type = c.type === 'directory' ? 'd' : '-';
          const perms = c.type === 'directory' ? 'rwxr-xr-x' : 'rw-r--r--';
          const size = c.type === 'file' ? String(c.content ? c.content.length : 0).padStart(6) : '   512';
          return `${type}${perms}  1 cyperuser cyperuser ${size} ${dateStr} ${c.name}${c.type === 'directory' ? '/' : ''}`;
        });
        termPrint('total ' + allItems.length + '\n' + lines.join('\n'));
      } else {
        const out = allItems.map(c => c.type === 'directory' ? c.name + '/' : c.name).join('  ');
        termPrint(out, 'term-line-success');
      }
      break;
    }

    case 'pwd':
      termPrint(currentPath.join('/'), 'term-line-success');
      break;

    case 'cd': {
      if (!args[0] || args[0] === '~') { currentPath = ['~']; updatePrompt(); break; }
      if (args[0] === '/') { currentPath = ['~']; updatePrompt(); termPrint('/', 'term-line-success'); break; }
      if (args[0] === '..') {
        if (currentPath.length > 1) currentPath.pop();
        updatePrompt(); break;
      }
      const dir = getCurrentDir();
      const target = dir && dir.children && dir.children.find(c => c.name === args[0] && c.type === 'directory');
      if (target) { currentPath.push(args[0]); updatePrompt(); }
      else termPrint(`cd: no such file or directory: ${args[0]}`, 'term-line-error');
      break;
    }

    case 'mkdir': {
      if (!args[0]) { termPrint('mkdir: missing operand', 'term-line-error'); break; }
      const nested = args[0] === '-p';
      const dirName = nested ? args[1] : args[0];
      if (!dirName) { termPrint('mkdir: missing directory name', 'term-line-error'); break; }

      if (nested && dirName.includes('/')) {
        // mkdir -p a/b/c — create nested dirs
        const parts = dirName.split('/');
        let node = getCurrentDir();
        let created = [];
        for (const part of parts) {
          if (!part) continue;
          let existing = node.children.find(c => c.name === part && c.type === 'directory');
          if (!existing) {
            const newDir = { name: part, type: 'directory', children: [] };
            node.children.push(newDir);
            created.push(part);
            node = newDir;
          } else { node = existing; }
        }
        termPrint(`Created: ${dirName}`, 'term-line-success');
        renderFileManager();
        break;
      }

      const dir = getCurrentDir();
      if (dir.children.find(c => c.name === dirName)) {
        termPrint(`mkdir: cannot create directory '${dirName}': File exists`, 'term-line-error'); break;
      }
      dir.children.push({ name: dirName, type: 'directory', children: [] });
      termPrint(`Directory '${dirName}' created successfully.`, 'term-line-success');
      renderFileManager();
      break;
    }

    case 'rmdir': {
      if (!args[0]) { termPrint('rmdir: missing operand', 'term-line-error'); break; }
      const dir = getCurrentDir();
      const target = dir.children.find(c => c.name === args[0] && c.type === 'directory');
      if (!target) { termPrint(`rmdir: '${args[0]}': No such directory`, 'term-line-error'); break; }
      if (target.children && target.children.length > 0) { termPrint(`rmdir: '${args[0]}': Directory not empty (use rm -r to force)`, 'term-line-error'); break; }
      dir.children.splice(dir.children.indexOf(target), 1);
      termPrint(`Directory '${args[0]}' removed.`, 'term-line-success');
      renderFileManager();
      break;
    }

    case 'touch': {
      if (!args[0]) { termPrint('touch: missing file operand', 'term-line-error'); break; }
      const dir = getCurrentDir();
      const now = new Date().toLocaleString();

      // touch -r file1 file2 → copy timestamp
      if (args[0] === '-r') {
        const src = dir.children.find(c => c.name === args[1]);
        const dst = dir.children.find(c => c.name === args[2]);
        if (!src) { termPrint(`touch: '${args[1]}': No such file`, 'term-line-error'); break; }
        if (!dst) { termPrint(`touch: '${args[2]}': No such file`, 'term-line-error'); break; }
        dst.accessed = src.accessed; dst.modified = src.modified;
        termPrint(`Timestamp of '${args[2]}' set from '${args[1]}'.`, 'term-line-success'); break;
      }

      // touch -d "date" file
      if (args[0] === '-d') {
        const dateStr = args[1]; const fname = args[2];
        if (!fname) { termPrint('touch: missing file name', 'term-line-error'); break; }
        let f = dir.children.find(c => c.name === fname);
        if (!f) { f = { name: fname, type: 'file', content: '', accessed: dateStr, modified: dateStr }; dir.children.push(f); renderFileManager(); }
        else { f.accessed = dateStr; f.modified = dateStr; }
        termPrint(`Timestamp of '${fname}' set to: ${dateStr}`, 'term-line-success'); break;
      }

      // touch -c file → update time only, no create
      if (args[0] === '-c') {
        const f = dir.children.find(c => c.name === args[1]);
        if (!f) { termPrint(`touch: '${args[1]}': File does not exist (skipped)`, 'term-line-output'); break; }
        f.accessed = now; f.modified = now;
        termPrint(`Timestamps updated: ${args[1]}`, 'term-line-success'); break;
      }

      // touch -a file → update access time only
      if (args[0] === '-a') {
        const fname = args[1];
        if (!fname) { termPrint('touch: missing file name', 'term-line-error'); break; }
        let f = dir.children.find(c => c.name === fname);
        if (!f) { f = { name: fname, type: 'file', content: '', accessed: now, modified: '' }; dir.children.push(f); renderFileManager(); }
        else { f.accessed = now; }
        termPrint(`Access time updated: ${fname}`, 'term-line-success'); break;
      }

      // touch -m file → update modified time only
      if (args[0] === '-m') {
        const fname = args[1];
        if (!fname) { termPrint('touch: missing file name', 'term-line-error'); break; }
        let f = dir.children.find(c => c.name === fname);
        if (!f) { f = { name: fname, type: 'file', content: '', accessed: '', modified: now }; dir.children.push(f); renderFileManager(); }
        else { f.modified = now; }
        termPrint(`Modified time updated: ${fname}`, 'term-line-success'); break;
      }

      // touch file → normal create
      const fname = args[0];
      if (!dir.children.find(c => c.name === fname)) {
        dir.children.push({ name: fname, type: 'file', content: '', accessed: now, modified: now });
        termPrint(`File '${fname}' created.`, 'term-line-success');
        renderFileManager();
      } else {
        // file exists — update both timestamps
        const f = dir.children.find(c => c.name === fname);
        f.accessed = now; f.modified = now;
        termPrint(`Timestamps updated: ${fname}`, 'term-line-success');
      }
      break;
    }

    case 'rm': {
      const flags   = args.filter(a => a.startsWith('-'));
      const isR     = flags.some(f => f.includes('r'));
      const isI     = flags.some(f => f.includes('i'));
      const isV     = flags.some(f => f.includes('v'));
      const isD     = flags.some(f => f === '-d');
      const targetName = args.find(a => !a.startsWith('-'));
      if (!targetName) { termPrint('rm: missing operand', 'term-line-error'); break; }
      const dir = getCurrentDir();
      const idx = dir.children.findIndex(c => c.name === targetName);
      if (idx === -1) { termPrint(`rm: cannot remove '${targetName}': No such file or directory`, 'term-line-error'); break; }
      const item = dir.children[idx];

      if (item.type === 'directory' && !isR && !isD) {
        termPrint(`rm: cannot remove '${targetName}': Is a directory (use rm -r)`, 'term-line-error'); break;
      }
      if (isD && item.type === 'directory' && item.children && item.children.length > 0) {
        termPrint(`rm: cannot remove '${targetName}': Directory not empty`, 'term-line-error'); break;
      }
      if (isI) {
        const confirmed = confirm(`rm: remove '${targetName}'?`);
        if (!confirmed) { termPrint(`rm: '${targetName}' not removed.`, 'term-line-output'); break; }
      }
      dir.children.splice(idx, 1);
      if (isV) termPrint(`removed '${targetName}'`, 'term-line-success');
      else termPrint(`Removed '${targetName}'${isR && item.type==='directory' ? ' and all contents' : ''}.`, 'term-line-success');
      renderFileManager();
      break;
    }

    case 'stat': {
      if (!args[0]) { termPrint('stat: missing file operand', 'term-line-error'); break; }
      const dir = getCurrentDir();
      const f = dir.children.find(c => c.name === args[0]);
      if (!f) { termPrint(`stat: cannot stat '${args[0]}': No such file or directory`, 'term-line-error'); break; }
      const now = new Date().toLocaleString();
      const size = f.type === 'file' ? (f.content ? f.content.length : 0) : 4096;
      termPrint(`  File: ${f.name}
  Size: ${size} bytes
  Type: ${f.type === 'directory' ? 'directory' : 'regular file'}
Access: ${f.accessed || now}
Modify: ${f.modified || now}
Change: ${now}
 Perms: ${f.type === 'directory' ? 'drwxr-xr-x' : '-rw-r--r--'}
 Owner: cyperuser`, 'term-line-success');
      break;
    }

    case 'history': {
      if (termHistory.length === 0) { termPrint('(no history)', 'term-line-output'); break; }
      const hist = [...termHistory].reverse().map((c, i) => `  ${String(i+1).padStart(3)}  ${c}`).join('\n');
      termPrint(hist);
      break;
    }

    case 'cat': {
      if (!args[0]) { termPrint('cat: missing file operand', 'term-line-error'); break; }
      const dir = getCurrentDir();
      const file = dir.children.find(c => c.name === args[0] && c.type === 'file');
      if (!file) { termPrint(`cat: ${args[0]}: No such file`, 'term-line-error'); break; }
      termPrint(file.content || '(empty file)');
      break;
    }

    case 'echo': {
      // echo "text" > file.txt
      const rIdx = args.indexOf('>');
      if (rIdx !== -1) {
        const text    = args.slice(0, rIdx).join(' ').replace(/^"|"$/g, '');

    case 'whoami':
      termPrint('cyperWeb — Created by CYPER WEB Team', 'term-line-success');
      break;

    case 'ps':
      termPrint(`PID  NAME              CPU%   MEM

      termPrint(`              total        used        free
Mem:           4096MB       1240MB      2856MB

document.getElementById('termInput').addEventListener('keydown', function(e) {
  if (e.key === 'Enter') {
    handleCommand(this.value);
    this.value = '';
  } else if (e.key === 'ArrowUp') {
    historyIndex = Math.min(historyIndex + 1, termHistory.length - 1);
    if (termHistory[historyIndex]) this.value = termHistory[historyIndex];
    e.preventDefault();
  } else if (e.key === 'ArrowDown') {
    historyIndex = Math.max(historyIndex - 1, -1);
    this.value = historyIndex >= 0 ? termHistory[historyIndex] : '';
    e.preventDefault();
  }
});

let fileNavPath = ['~'];
let selectedFile = null;

function renderFileManager() {
  const dir = getDir(fileNavPath);
  if (!dir) return;
  document.getElementById('filePath').value = fileNavPath.join('/');
  const grid = document.getElementById('fileGrid');
  grid.innerHTML = '';
  if (!dir.children || dir.children.length === 0) {
    grid.innerHTML = '<div style="color:var(--text-dim);font-size:12px;padding:20px">Empty folder</div>';
    return;
  }
  dir.children.forEach(item => {
    const el = document.createElement('div');
    el.className = 'file-item';
    const icon = item.type === 'directory' ? '📁' : getFileIcon(item.name);
    el.innerHTML = `<span class="fi-icon">${icon}</span><span class="fi-name">${item.name}</span>`;
    el.addEventListener('click', () => {
      document.querySelectorAll('.file-item').forEach(x => x.classList.remove('selected'));
      el.classList.add('selected');
      selectedFile = item;
    });
    el.addEventListener('dblclick', () => {
      if (item.type === 'directory') {
        fileNavPath.push(item.name);
        renderFileManager();
      } else {
        openEditorWith(item);
      }
    });
    grid.appendChild(el);
  });
}

function getFileIcon(name) {
  const ext = name.split('.').pop().toLowerCase();
  const icons = { txt: '📄', js: '📜', html: '🌐', css: '🎨', json: '📋', md: '📝', py: '🐍', sh: '⚙️' };
  return icons[ext] || '📄';
}

  if (fileNavPath.length > 1) { fileNavPath.pop(); renderFileManager(); }
}
function fileNavHome() { fileNavPath = ['~']; renderFileManager(); }

function fileNewFolder() {
  const name = prompt('New folder name:');
  if (!name || !name.trim()) return;
  const dir = getDir(fileNavPath);
  if (dir.children.find(c => c.name === name.trim())) {
    showNotif('File Manager', `'${name}' already exists!`, true); return;
  }
  dir.children.push({ name: name.trim(), type: 'directory', children: [] });
  showNotif('File Manager', `Folder '${name}' created.`);
  renderFileManager();
}

function fileDeleteSelected() {
  if (!selectedFile) { showNotif('File Manager', 'Select a file or folder first.', true); return; }
  const confirmed = confirm(`Delete '${selectedFile.name}'?${selectedFile.type === 'directory' ? '\nThis will delete all contents inside.' : ''}`);
  if (!confirmed) return;
  const dir = getDir(fileNavPath);
  const idx = dir.children.findIndex(c => c.name === selectedFile.name);
  if (idx !== -1) { dir.children.splice(idx, 1); }
  showNotif('File Manager', `'${selectedFile.name}' deleted.`);
  selectedFile = null;
  renderFileManager();
}

function editorSave() {
  const name = document.getElementById('editorFilename').value;
  const content = document.getElementById('editorArea').value;
  if (editorCurrentFile) {
    editorCurrentFile.content = content;
    editorCurrentFile.name = name;
    showNotif('Editor', `Saved: ${name}`, false);
  } else {
    const dir = getCurrentDir();
    const existing = dir.children.find(c => c.name === name);
    if (existing) { existing.content = content; }
    else dir.children.push({ name, type: 'file', content });
    showNotif('Editor', `Saved: ${name}`, false);
  }
  renderFileManager();
}

  const col = before.split('\n').pop().length + 1;

let zTop = 10;
function openWindow(id) {
  const w = document.getElementById(id);
  w.classList.add('open');
  w.style.zIndex = ++zTop;
  document.querySelectorAll('.window').forEach(x => x.classList.remove('focused'));
  w.classList.add('focused');
  updateTaskbar();
  if (id === 'fileWindow') renderFileManager();
  if (id === 'terminalWindow') setTimeout(() => document.getElementById('termInput').focus(), 100);
}

function closeWindow(id) {
  document.getElementById(id).classList.remove('open');
  updateTaskbar();
}

function minimizeWindow(id) {
  document.getElementById(id).classList.remove('open');
  updateTaskbar();
}

function maximizeWindow(id) {
  const w = document.getElementById(id);
  if (w.style.width === '100%') {
    w.style.width = '';
    w.style.height = '';
    w.style.top = '';
    w.style.left = '';
  } else {
    w.style.width = '100%';
    w.style.height = `calc(100vh - ${getComputedStyle(document.documentElement).getPropertyValue('--taskbar-h')})`;
    w.style.top = '0';
    w.style.left = '0';
    w.style.borderRadius = '0';
  }
}

function updateTaskbar() {
  ['terminal','files','sysmon','editor'].forEach(k => {
    const map = { terminal:'terminalWindow', files:'fileWindow', sysmon:'sysmonWindow', editor:'editorWindow' };
    const btn = document.getElementById('tb-' + k);
    btn.classList.toggle('active', document.getElementById(map[k]).classList.contains('open'));
  });
}

// Drag windows
document.querySelectorAll('.window-titlebar').forEach(tb => {
  let dragging = false, ox, oy;
  tb.addEventListener('mousedown', e => {
    if (e.target.classList.contains('wc')) return;
    const win = tb.closest('.window');
    win.style.zIndex = ++zTop;
    document.querySelectorAll('.window').forEach(x => x.classList.remove('focused'));
    win.classList.add('focused');
    dragging = true;
    const rect = win.getBoundingClientRect();
    ox = e.clientX - rect.left;
    oy = e.clientY - rect.top;
    e.preventDefault();
  });
  document.addEventListener('mousemove', e => {
    if (!dragging) return;
    const win = tb.closest('.window');
    const dw = window.innerWidth;
    const dh = window.innerHeight - 44;
    let x = e.clientX - ox;
    let y = e.clientY - oy;

    // Snap to edges (within 12px)
    if (x < 12) x = 0;
    if (y < 12) y = 0;
    if (x + win.offsetWidth > dw - 12) x = dw - win.offsetWidth;
    if (y + win.offsetHeight > dh - 12) y = dh - win.offsetHeight;

    x = Math.max(0, Math.min(x, dw - win.offsetWidth));
    y = Math.max(0, Math.min(y, dh - win.offsetHeight));
    win.style.left = x + 'px';
    win.style.top = y + 'px';
  });
  document.addEventListener('mouseup', () => { dragging = false; });
});

  });
  document.addEventListener('mouseup', () => { resizing = false; });
});

// DESKTOP WALLPAPER — CYBER ART
// ==============================
function drawWallpaper() {
  const canvas = document.getElementById('desktopWallpaper');
  const desktop = document.getElementById('desktop');
  canvas.width  = desktop.offsetWidth;
  canvas.height = desktop.offsetHeight;
  const ctx = canvas.getContext('2d');
  const W = canvas.width, H = canvas.height;
  const cx = W / 2, cy = H / 2;

  // ── PURE BLACK ──
  ctx.fillStyle = '#000';
  ctx.fillRect(0, 0, W, H);

  // ── DEEP ATMOSPHERE ──
  const atmo = ctx.createRadialGradient(cx, cy, 0, cx, cy, H * 0.85);
  atmo.addColorStop(0,   'rgba(0, 20, 8, 0.95)');
  atmo.addColorStop(0.5, 'rgba(0, 8, 3, 0.7)');
  atmo.addColorStop(1,   '#000');
  ctx.fillStyle = atmo;
  ctx.fillRect(0, 0, W, H);

  // ── GRID PERSPECTIVE ──
  const gridAlpha = 0.07;
  ctx.strokeStyle = `rgba(0,255,65,${gridAlpha})`;
  ctx.lineWidth = 1;
  // Horizontal lines vanishing to center
  for (let i = 0; i <= 20; i++) {
    const y = H * 0.55 + (H * 0.45 * i / 20);
    const spread = (i / 20);
    ctx.beginPath();
    ctx.moveTo(cx - W * 0.8 * spread, y);
    ctx.lineTo(cx + W * 0.8 * spread, y);
    ctx.stroke();
  }
  // Vertical lines vanishing to center
  for (let i = -12; i <= 12; i++) {
    ctx.beginPath();
    ctx.moveTo(cx + (W * 0.06) * i, H * 0.55);
    ctx.lineTo(cx + (W * 0.5) * (i / 12), H);
    ctx.stroke();
  }

  // ── HEXAGON PATTERN — center ──
  function hexagon(x, y, r, lineW, color, alpha) {
    ctx.beginPath();
    for (let a = 0; a < 6; a++) {
      const angle = (Math.PI / 3) * a - Math.PI / 6;
      const px = x + r * Math.cos(angle);
      const py = y + r * Math.sin(angle);
      a === 0 ? ctx.moveTo(px, py) : ctx.lineTo(px, py);
    }
    ctx.closePath();
    ctx.strokeStyle = color || `rgba(0,255,65,${alpha || 0.15})`;
    ctx.lineWidth = lineW || 1;
    ctx.stroke();
  }

  // Outer rings of hexagons
  const hexR  = Math.min(W, H) * 0.045;
  const rings = [
    { count: 1,  radius: 0,                  alpha: 0.5,  lw: 1.5 },
    { count: 6,  radius: hexR * 2.1,         alpha: 0.3,  lw: 1   },
    { count: 12, radius: hexR * 4.2,         alpha: 0.15, lw: 0.7 },
    { count: 18, radius: hexR * 6.3,         alpha: 0.08, lw: 0.5 },
  ];

  rings.forEach(({ count, radius, alpha, lw }) => {
    for (let i = 0; i < count; i++) {
      const angle = (Math.PI * 2 / count) * i;
      const hx = cx + radius * Math.cos(angle);
      const hy = cy * 0.72 + radius * Math.sin(angle);
      hexagon(hx, hy, hexR, lw, null, alpha);
    }
  });

  // ── GLOWING CENTER CIRCLE ──
  const centerGlow = ctx.createRadialGradient(cx, cy*0.72, 0, cx, cy*0.72, hexR*7);
  centerGlow.addColorStop(0,   'rgba(0,255,65,0.18)');
  centerGlow.addColorStop(0.3, 'rgba(0,255,65,0.06)');
  centerGlow.addColorStop(0.7, 'rgba(180,0,0,0.04)');
  centerGlow.addColorStop(1,   'transparent');
  ctx.fillStyle = centerGlow;
  ctx.fillRect(0, 0, W, H);

  // ── CONCENTRIC CIRCLES ──
  for (let r = hexR * 2; r <= hexR * 14; r += hexR * 2.5) {
    ctx.beginPath();
    ctx.arc(cx, cy * 0.72, r, 0, Math.PI * 2);
    ctx.strokeStyle = `rgba(0,255,65,${0.04 + (hexR * 2) / r * 0.06})`;
    ctx.lineWidth = 0.5;
    ctx.stroke();
  }

  // ── CORNER BRACKETS ──
  const bSize = Math.min(W, H) * 0.07;
  const bGap  = Math.min(W, H) * 0.03;
  const corners = [
    [bGap,     bGap,      1,  1 ],
    [W-bGap,   bGap,     -1,  1 ],
    [bGap,     H-bGap,    1, -1 ],
    [W-bGap,   H-bGap,   -1, -1 ],
  ];
  ctx.strokeStyle = 'rgba(0,255,65,0.6)';
  ctx.lineWidth   = 1.5;
  ctx.shadowColor = '#00ff41';
  ctx.shadowBlur  = 8;
  corners.forEach(([x, y, dx, dy]) => {
    ctx.beginPath();
    ctx.moveTo(x + dx * bSize, y);
    ctx.lineTo(x, y);
    ctx.lineTo(x, y + dy * bSize);
    ctx.stroke();
  });
  ctx.shadowBlur = 0;

  // ── SCAN LINES ──
  for (let y = 0; y < H; y += 3) {
    ctx.fillStyle = 'rgba(0,0,0,0.1)';
    ctx.fillRect(0, y, W, 1);
  }

  // ── NEON BORDER ──
  const bw = 2;
  const mkG = (x0,y0,x1,y1) => {
    const g = ctx.createLinearGradient(x0,y0,x1,y1);
    g.addColorStop(0,   'rgba(0,255,65,0)');
    g.addColorStop(0.2, 'rgba(0,255,65,0.8)');
    g.addColorStop(0.5, '#00ff41');
    g.addColorStop(0.8, 'rgba(0,255,65,0.8)');
    g.addColorStop(1,   'rgba(0,255,65,0)');
    return g;
  };
  ctx.shadowColor = '#00ff41'; ctx.shadowBlur = 10;
  ctx.fillStyle = mkG(0,0,W,0); ctx.fillRect(0,    0,    W,  bw);
  ctx.fillStyle = mkG(0,0,W,0); ctx.fillRect(0,    H-bw, W,  bw);
  ctx.fillStyle = mkG(0,0,0,H); ctx.fillRect(0,    0,    bw, H);
  ctx.fillStyle = mkG(0,0,0,H); ctx.fillRect(W-bw, 0,    bw, H);
  ctx.shadowBlur = 0;

  // ── WELCOME TEXT ──
  const fs = Math.max(11, Math.floor(W / 85));
  ctx.font      = `${fs}px 'JetBrains Mono', monospace`;
  ctx.textAlign = 'center';
  ctx.fillStyle = 'rgba(0,255,65,0.5)';
  ctx.shadowColor = '#00ff41'; ctx.shadowBlur = 8;
  ctx.fillText('Welcome Eng Karim & Eng Mohamed', W/2, H - 14);
  ctx.shadowBlur = 0;
}
